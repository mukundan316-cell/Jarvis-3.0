// Dramatic failover simulation
export function DisasterRecoveryDemo() {
  const [isDisaster, setIsDisaster] = useState(false);
  const [recoveryProgress, setRecoveryProgress] = useState(0);
  
  const simulateDisaster = async () => {
    setIsDisaster(true);
    // Show red alerts, system "failing"
    await sleep(2000);
    
    // Begin recovery with progress animation
    for (let i = 0; i <= 100; i += 10) {
      setRecoveryProgress(i);
      await sleep(200);
    }
    setIsDisaster(false);
  };
  
  return (
    <div className={isDisaster ? "animate-pulse bg-red-900/20" : ""}>
      <Button variant="destructive" onClick={simulateDisaster}>
        ‚ö†Ô∏è Simulate System Failure
      </Button>
      {isDisaster && (
        <Alert className="mt-4 border-red-500">
          <AlertTitle>System Failure Detected!</AlertTitle>
          <Progress value={recoveryProgress} />
          <p>Auto-recovery in progress... {recoveryProgress}%</p>
        </Alert>
      )}
    </div>
  );
}
