```typescript
// Demo controller service
class DemoController {
  private scenarios = new Map([
    ['apex_manufacturing', {
      // Complete scenario data with real files
      broker: 'Willis Towers Watson',
      files: ['demo_acord_125.pdf', 'demo_sov.xlsx', 'demo_loss_runs.pdf'],
      metadata: { tiv: 15000000, occupancy: 'Manufacturing', address: '123 Industrial Blvd' }
    }],
    ['retail_complex', {
      broker: 'Marsh & McLennan', 
      files: ['retail_acord.pdf', 'retail_sov.xlsx'],
      metadata: { tiv: 8500000, occupancy: 'Retail', address: '456 Main Street' }
    }]
  ]);

  async triggerDemoEmail(scenarioId: string) {
    const scenario = this.scenarios.get(scenarioId);
    
    // Show email notification
    notificationService.show({
      type: 'email',
      message: `New email from ${scenario.broker}`,
      duration: 3000
    });
    
    // Delay for realism
    await new Promise(resolve => setTimeout(resolve, 2000));
    
    // Create submission (same as real email flow)
    const submission = await submissionService.create({
      source: 'email_simulation',
      scenario: scenarioId,
      ...scenario
    });
    
    // Trigger UI updates
    eventBus.emit('new_submission', submission);
    
    return submission;
  }
}
```
